#!/bin/bash
#
#Kevin Blum



main()
{


    x=1
    y=$1
    until ((x>y))
    do
        if ((x % 5 == 0))
        then
            echo "*$x"
        else
            echo "$x"
        fi
        ((x++))
    done

    while ((y>0))
    do
        if ((y % 5 == 0))
        then
            echo "*$y"
        else
            echo "$y"
        fi
        ((y--))
    done        
    

}

main "$1"
