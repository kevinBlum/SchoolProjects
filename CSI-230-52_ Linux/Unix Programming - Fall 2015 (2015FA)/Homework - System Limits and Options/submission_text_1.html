void openfile(){

    int fd;
    int maxpath;
    errno = 0;
    char* filename;
    printf("What is the name of the file?");
    scanf("%s",filename);
    fd = open(filename, O_WRONLY | O_CREAT);
    if (fd == -1) {
        perror ("The following error occurred");
    }
    if ((maxpath = fpathconf(fd, _PC_PATH_MAX)) == -1){
        if (errno == 0) {
            printf("There is no limit");
        }
        else if (errno > 0){
            perror ("The following error occurred");
        }
        else {
            printf("The max path is %d", maxpath);
            close(fd);
        }
    }

}
